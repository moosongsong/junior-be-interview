    }
}

class Subscriber implements Observer {
    private String name;
    private Subject alarm;
    
    public Subscriber(String name, Subject alarm) {
        this.name = name;
        this.alarm = alarm;
    }
    
    @Override
    public void update() {
        String msg = (String)alarm.getUpdate(this);
        System.out.println(name + " got alarm : "+ msg);
    }
}
